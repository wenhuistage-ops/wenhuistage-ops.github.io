### 當前實現（第 224-307 行）

```javascript
function punch(sessionToken, type, lat, lng, note) {
```

### 缺陷詳情

| 檢查項 | 應該有 | 實際 | 風險 |
|--------|--------|------|------|
| 檢查當天是否已打卡進 | ✅ | ❌ | 可能重複打卡上班 |
| 檢查當天是否已打卡出 | ✅ | ❌ | 可能重複打卡下班 |
| 驗證上班時間在下班之前 | ✅ | ❌ | 時間邏輯混亂 |
| 防止用戶連續點擊 | ✅ | ⚠️ 前端做了 | 後端無防護 |

### 重現步驟

```
1. 用戶上午 9:00 打卡上班 ✅
2. 用戶因網絡延遲，再次點擊打卡上班 ✅ 成功（不應該！）
3. 結果：同一天有 2 條「上班」記錄
```

### 改善建議

```javascript
// ✅ 改進版本
function punch(sessionToken, type, lat, lng, note) {
  // ... 驗證代碼 ...

  const today = getDateString(new Date());
  
  const existingRecords = getRecordsForDate(user.userId, today);

  if (alreadyExists) {

  // 🌟 新增：驗證時間邏輯
  const punchInTime = findRecordType(existingRecords, '上班');
  if (type === '下班' && !punchInTime) {
    return { 
      ok: false, 
      code: "ERR_MUST_PUNCH_IN_FIRST"
    };
  }
  
  // ... 寫入記錄 ...
```

## 🔴 問題 7.2：補卡申請冪等性 - **發現缺陷**

### 當前實現（第 675-695 行）

```javascript
function updateReviewStatus(rowNumber, status, note) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_ATTENDANCE);
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const reviewStatusCol = headers.indexOf('管理員審核') + 1;
    
    // ❌ 直接更新，無冪等性檢查
    sheet.getRange(rowNumber, reviewStatusCol).setValue(status);
    
    return { ok: true, msg: "審核成功" };
  } catch (err) {
    return { ok: false, msg: `審核失敗：${err.message}` };
  }
}
```

### 缺陷詳情

| 檢查項 | 應該有 | 實際 | 風險 |
|--------|--------|------|------|
| 檢查申請是否已批准 | ✅ | ❌ | 重複批准 |
| 檢查是否會添加重複記錄 | ✅ | ❌ | 補打卡重複 |
| 記錄批准者和時間 | ✅ | ❌ | 無審計跡 |
| 防止已拒絕申請被批准 | ✅ | ❌ | 狀態混亂 |

### 重現步驟

```
1. 管理員批准補卡申請 #5426 ✅
2. 管理員因為沒看清，再次點擊批准 ✅ 成功（不應該！）
3. 結果：補打卡記錄可能被添加 2 次
```

### 改善建議

```javascript
// ✅ 改進版本
function updateReviewStatus(rowNumber, status, note) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_ATTENDANCE);
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const reviewStatusCol = headers.indexOf('管理員審核') + 1;
    
    // 🌟 新增：讀取當前狀態
    const currentStatus = sheet.getRange(rowNumber, reviewStatusCol).getValue();
    
    // 🌟 新增：冪等性檢查
    if (currentStatus === 'v') {
      return { 
        ok: true, 
        msg: "申請已批准（冪等操作）",
        idempotent: true
      };
    }
    
    if (currentStatus === 'x') {
      return { 
        ok: false, 
        msg: "申請已被拒絕，不可批准"
      };
    }
    
    // 如果是補打卡，先檢查是否已添加記錄
    if (status === 'v' && isAdjustmentRecord(sheet, rowNumber)) {
      const existingPunch = findPunchRecord(sheet, rowNumber);
      if (existingPunch) {
        return { 
          ok: false, 
          msg: "該補打卡已添加，不可重複批准"
        };
      }
      
      // 🌟 才添加打卡記錄
      addPunchFromAdjustmentRecord(sheet, rowNumber);
    }
    
    // 🌟 新增：記錄批准者和時間
    sheet.getRange(rowNumber, reviewStatusCol).setValue(status);
    const approverCol = headers.indexOf('批准者') + 1;
    const approveTimeCol = headers.indexOf('批准時間') + 1;
    
    if (approverCol > 0) {
      sheet.getRange(rowNumber, approverCol).setValue(getCurrentUserId());
    }
    if (approveTimeCol > 0) {
      sheet.getRange(rowNumber, approveTimeCol).setValue(new Date());
    }
    
    return { ok: true, msg: "審核成功" };
  } catch (err) {
    return { ok: false, msg: `審核失敗：${err.message}` };
  }
}
```
