- [ ] **Session 清理機制**

    ```

### 1.2 預期改善

```javascript
// 後端應該實現
const SESSION_TTL_HOURS = 48;  // Session 有效期 48 小時

function checkSession(token) {
    const session = getSessionFromDB(token);
    
    if (!session) {
        return { ok: false, code: 'SESSION_INVALID' };
    }
    
    // 檢查過期時間
    const expirationTime = new Date(session.createdAt);
    expirationTime.setHours(expirationTime.getHours() + SESSION_TTL_HOURS);
    
    if (new Date() > expirationTime) {
        deleteSession(token);  // 清除過期 session
        return { ok: false, code: 'SESSION_EXPIRED' };
    }
    
    // 可選：更新 session 的最後訪問時間，實現滑動過期
    session.lastAccessedAt = new Date();
    saveSession(session);
    
    return { ok: true, user: session.user };
}
```

## 2. 問題 5.1：缺少輸入驗證 ⚠️

**狀態**：❌ 未確認

**描述**：GAS 後端可能未驗證前端傳來的參數，存在注入攻擊風險

### 2.1 關鍵檢查點

- [ ] **Action 參數驗證**
  - 是否白名單驗證所有允許的 action？
  - 範例應該檢查的 action：
    ```javascript
    const VALID_ACTIONS = [
        'getLoginUrl',
        'getProfile',
        'checkSession',
        'addAttendance',
        'getMonthlyAttendance',
        'requestAdjustment',
        // ... 其他 action
    ];
    
    if (!VALID_ACTIONS.includes(action)) {
        return { ok: false, code: 'INVALID_ACTION' };
    }
    ```

- [ ] **用戶輸入驗證**
  - userId：是否驗證格式和長度？
  - date：是否驗證日期格式？
  - month：是否驗證 YYYY-MM 格式？
  - location：是否驗證座標範圍？
  - adjustment reason：是否檢查長度限制？

**檢查樣本代碼**：
```javascript
// ❌ 危險的實現
function doGet(e) {
    const action = e.parameter.action;
    const userId = e.parameter.userId;
    
    // 直接執行 SQL 查詢（容易被注入）
    const result = sheet.getRange(`A:F`).getValues()
        .filter(row => row[0] === userId);  // ❌ 無驗證
}

// ✅ 安全的實現
function doGet(e) {
    const action = e.parameter.action;
    const userId = e.parameter.userId;
    
    // 驗證 action
    if (!VALID_ACTIONS.includes(action)) {
        return ContentService.createTextOutput(
            JSON.stringify({ ok: false, code: 'INVALID_ACTION' })
        ).setMimeType(ContentService.MimeType.JSON);
    }
    
    // 驗證 userId 格式
    if (!isValidUserId(userId)) {
        return ContentService.createTextOutput(
            JSON.stringify({ ok: false, code: 'INVALID_USER_ID' })
        ).setMimeType(ContentService.MimeType.JSON);
    }
    
    // ... 安全的業務邏輯
}
```

### 2.2 需要檢查的所有參數

| 參數 | 來源 | 驗證方式 | 檢查項 |
|------|------|--------|--------|
| action | query | 白名單 | 必須在 VALID_ACTIONS 中 |
| token/sessionToken | query/cookie | 長度+存在性 | 長度 32-256 字符 |
| userId | query | 正則表達式 | 格式驗證 |
| otoken | query | 長度限制 | 最大 1000 字符 |
| date | query | 日期格式 | YYYY-MM-DD |
| month | query | 日期格式 | YYYY-MM |
| latitude | query | 數字範圍 | -90 到 90 |
| longitude | query | 數字範圍 | -180 到 180 |
| reason | query | 長度限制 | 最多 500 字符 |
| location | query | 地點 ID 驗證 | 必須是已註冊的地點 |

## 4. 問題 7.1：缺少打卡記錄去重檢查 ⚠️

**狀態**：❌ 未確認

**描述**：快速雙擊打卡按鈕可能產生重複記錄

### 4.1 檢查項

- [ ] **打卡去重機制**
  - 後端是否檢查同一用戶在同一天是否已經打卡進/打卡出？
  - ```javascript
    function handlePunchIn(userId, punchTime) {
        // 檢查當天是否已有打卡進記錄
        const todayRecords = getRecordsForDate(userId, getTodayDateString());
        const alreadyPunchedIn = todayRecords.some(r => r.type === 'in');
        
        if (alreadyPunchedIn) {
            return { ok: false, code: 'ALREADY_PUNCHED_IN' };
        }
        
        // 記錄打卡
        return addAttendanceRecord(userId, 'in', punchTime);
    }
    ```

- [ ] **時間窗口驗證**
  - 是否檢查打卡時間的邏輯？
  - 打卡出的時間應該晚於打卡進的時間
  - ```javascript
    if (record.punchOut <= record.punchIn) {
        return { ok: false, code: 'INVALID_PUNCH_TIME' };
    }
    ```

- [ ] **重複提交驗證**
  - 如果客戶端在 5 秒内重複傳送相同的請求？
  - 是否有防止重複提交的機制（例如：請求 ID）？

### 4.2 預期改善

```javascript
// 後端應該實現
function addAttendance(userId, punchType, location, gps) {
    // 驗證參數
    if (!isValidUserId(userId) || !['in', 'out'].includes(punchType)) {
        return { ok: false, code: 'INVALID_PARAMS' };
    }
    
    // 檢查去重
    const today = getTodayDateString();
    const todayRecords = getAttendanceRecords(userId, today);
    
    if (punchType === 'in' && todayRecords.some(r => r.type === 'in')) {
        return { ok: false, code: 'ALREADY_PUNCHED_IN' };
    }
    
    if (punchType === 'out' && !todayRecords.some(r => r.type === 'in')) {
        return { ok: false, code: 'MUST_PUNCH_IN_FIRST' };
    }
    
    // 記錄打卡
    const record = {
        userId: userId,
        type: punchType,
        timestamp: new Date(),
        location: location,
        gps: gps
    };
    
    saveRecord(record);
    
    return { ok: true, code: 'PUNCH_SUCCESS' };
}
```

## 5. 問題 7.2：補卡申請批准後未檢查重複 ⚠️

**狀態**：❌ 未確認

**描述**：管理員重複點擊"批准"可能產生重複打卡

### 5.1 檢查項

- [ ] **冪等性檢查（Idempotency）**
  - 批准補卡申請時，是否檢查申請是否已經被批准？
  - 是否防止重複添加相同的打卡記錄？

**檢查實現**：
```javascript
// ❌ 危險的實現 - 沒有冪等性檢查
function approveAdjustment(requestId) {
    const request = getAdjustmentRequest(requestId);
    
    // 直接添加記錄，不檢查是否已批准
    addAttendanceRecord(request.userId, request.punchType, request.punchTime);
    
    updateRequest(requestId, { status: 'approved' });
    return { ok: true };
}

// ✅ 安全的實現 - 有冪等性檢查
function approveAdjustment(requestId) {
    const request = getAdjustmentRequest(requestId);
    
    // 檢查申請狀態
    if (request.status === 'approved') {
        return { ok: true, code: 'ALREADY_APPROVED' };
    }
    
    if (request.status === 'rejected') {
        return { ok: false, code: 'REQUEST_ALREADY_REJECTED' };
    }
    
    // 檢查是否已有相同的記錄
    const existingRecord = getAttendanceRecord(
        request.userId, 
        request.date, 
        request.punchType
    );
    
    if (existingRecord) {
        return { ok: false, code: 'RECORD_ALREADY_EXISTS' };
    }
    
    // 添加打卡記錄
    addAttendanceRecord(request.userId, request.punchType, request.punchTime);
    
    // 更新申請狀態
    updateRequest(requestId, { 
        status: 'approved',
        approvedAt: new Date(),
        approvedBy: currentAdminUserId
    });
    
    return { ok: true, code: 'APPROVAL_SUCCESS' };
}
```
